Occupation et capacité quotidiennes

Occupation et capacité quotidiennes du réseau de refuges de Toronto, mis à jour chaque jour à partir des données ouvertes de la Ville de Toronto.

daily_occ = FileAttachment("../data/daily_occupancy.json").json()

function fmtDate(s) {
  const [y, m, d] = s.split("-").map(Number)
  return new Date(y, m - 1, d).toLocaleDateString("fr-CA", {
    year: "numeric", month: "long", day: "numeric"
  })
}

allDates = [...new Set(daily_occ.map(r => r.date))].sort().reverse()
latestDate = allDates[0]

La version BonQuery de ce tableau ajoute deux fonctions absentes de la page de la Ville. Le sélecteur de date ci-dessous permet de consulter toute date depuis janvier 2021 jusqu’à aujourd’hui ; la page Utilisation quotidienne des refuges et services d’hébergement de nuit de la Ville n’affiche que sa date de publication la plus récente. La colonne Hors service (calculée comme financé − réel) signale les lits, chambres ou espaces temporairement indisponibles — voir la page de validation des données pour une explication de ce que cela représente et pourquoi cela compte.

function getVal(rows, key) {
  const row = rows.find(r => r.key === key)
  return row ? row.ind : null
}

validationChecks = {
  const rows = latestRows
  const checks = []
  const expectedKeys = [
    "all_shelter","room_based","singles_sector","singles_shelter","allied_summ",
    "temp_summ","iso_summ","fam_total","fam_emerg","fam_trans_r","fam_hotel",
    "sng_hotel","singles_total","emerg_total","mix_emerg","men_emerg","wom_emerg",
    "yth_emerg","trans_total","mix_trans","fam_trans_b","men_trans","wom_trans",
    "yth_trans","allied_total","respites","dropin","temp_resp","hotels","iso"
  ]
  const presentKeys = new Set(rows.map(r => r.key))
  const missingKeys = expectedKeys.filter(k => !presentKeys.has(k))
  checks.push({
    name: "Toutes les lignes attendues présentes",
    pass: missingKeys.length === 0,
    detail: missingKeys.length > 0 ? `Manquantes : ${missingKeys.join(", ")}` : null
  })
  const allShelter = getVal(rows, "all_shelter")
  const compSum = ["fam_total","sng_hotel","singles_total","allied_total","temp_resp","hotels","iso"]
    .reduce((s, k) => s + (getVal(rows, k) ?? 0), 0)
  checks.push({
    name: "Le total Tous les programmes de refuge correspond aux composantes",
    pass: allShelter === compSum,
    detail: allShelter !== compSum
      ? `all_shelter.ind=${allShelter}, somme=${compSum}, diff=${allShelter - compSum}` : null
  })
  const singlesTotal = getVal(rows, "singles_total")
  const emergTotal   = getVal(rows, "emerg_total")
  const transTotal   = getVal(rows, "trans_total")
  checks.push({
    name: "Total secteurs individus : urgence + transition = total individus",
    pass: singlesTotal === (emergTotal + transTotal),
    detail: singlesTotal !== (emergTotal + transTotal)
      ? `individus=${singlesTotal}, urgence+transition=${emergTotal + transTotal}` : null
  })
  const capRows    = rows.filter(r => r.cap !== null && r.cap > 0)
  const arithFails = capRows.filter(r => Math.abs(r.occ + r.unocc - r.cap) > 0.5)
  checks.push({
    name: "Arithmétique interne : occupé + inoccupé = capacité réelle",
    pass: arithFails.length === 0,
    detail: arithFails.length > 0
      ? arithFails.map(r => `${r.key}: ${r.occ}+${r.unocc}≠${r.cap}`).join("; ") : null
  })
  return checks
}

function n(v)   { return v !== null && v !== undefined ? v.toLocaleString("fr-CA") : "—" }
function pct(v) { return v !== null && v !== undefined ? v.toFixed(1) + " %" : "—" }

function offlineCell(r) {
  if (r.cap === null || r.cap === undefined) return `<td></td>`
  if (r.cap_funding === null || r.cap_funding === undefined) return `<td></td>`
  const offline = r.cap_funding - r.cap
  if (offline <= 0) return `<td class="bq-num">${offline.toLocaleString("fr-CA")}</td>`
  return `<td class="bq-num bq-danger">${offline.toLocaleString("fr-CA")}</td>`
}

function tableRow(r) {
  const pad = (r.indent ?? 0) * 16
  const cls = r.is_total ? "bq-row-total" : "bq-row"
  const bld = r.is_total ? " bq-bold" : ""
  if (r.col_type === "summary") {
    return `<tr class="${cls}">
      <td style="padding-left:${pad}px" class="${bld}">${r.label}</td>
      <td class="bq-num${bld}">${n(r.ind)}</td>
      <td></td><td></td><td></td><td></td><td></td>
    </tr>`
  }
  return `<tr class="${cls}">
    <td style="padding-left:${pad}px" class="${bld}">${r.label}</td>
    <td class="bq-num${bld}">${n(r.ind)}</td>
    <td class="bq-num">${n(r.occ)}</td>
    <td class="bq-num">${n(r.unocc)}</td>
    <td class="bq-num">${n(r.cap)}</td>
    ${offlineCell(r)}
    <td class="bq-num">${pct(r.rate)}</td>
  </tr>`
}

Ce tableau reproduit le rapport d’utilisation quotidienne des refuges et services d’hébergement de nuit de la Ville de Toronto à partir des données ouvertes de la Ville. Le sous-total Tous les programmes de refuge et toutes les lignes de détail sont dérivés de l’exportation des données ouvertes CKAN. Les Programmes de passerelle et de triage sont suivis dans le système opérationnel de la Ville mais ne sont pas publiés dans les données ouvertes ; BonQuery collecte ce chiffre quotidiennement depuis la page du recensement des refuges de la Ville et l’affiche lorsqu’il est disponible. Le Total des personnes hébergées égale Tous les programmes de refuge + Passerelle et triage lorsque les deux sont disponibles. Certains chiffres de capacité et de lits inoccupés peuvent différer légèrement du tableau de bord en direct de la Ville en raison de différences dans la déclaration des lits hors service entre l’exportation CKAN et le système opérationnel de la Ville. Une comparaison côte à côte complète est disponible sur la page de validation des données.
